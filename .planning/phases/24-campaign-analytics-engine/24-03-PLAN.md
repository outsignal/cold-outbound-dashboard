<!-- API response shapes from Plan 02 -->

GET /api/analytics/campaigns response:
```typescript
{
  campaigns: Array<{
    id: string;           // Outsignal campaign cuid
    name: string;
    workspace: string;
    channels: string[];   // ["email"], ["linkedin"], ["email", "linkedin"]
    sent: number;
    replyRate: number;
    openRate: number;
    bounceRate: number;
    interestedRate: number;
    replied: number;
    opened: number;
    bounced: number;
    interested: number;
    copyStrategy: string | null;
    status: string;
  }>;
  total: number;
  period: string;
  filters: { workspace: string | null; sort: string; order: string };
}
```

GET /api/analytics/campaigns/[id]/steps response:
```typescript
{
  campaignId: string;
  campaignName: string;
  steps: Array<{
    step: number;
    channel: "email" | "linkedin";
    label: string;          // Subject line or LinkedIn action type
    sent: number;
    replied: number;
    replyRate: number;
    interestedCount: number;
    objectionCount: number;
    intentDistribution: Record<string, number>;
  }>;
}
```

GET /api/analytics/strategies response:
```typescript
{
  strategies: Array<{
    strategy: string;       // "creative-ideas" | "pvp" | "one-liner" | "custom" | "Unknown"
    campaignCount: number;
    avgReplyRate: number;
    avgOpenRate: number;
    avgBounceRate: number;
    avgInterestedRate: number;
    totalSent: number;
    totalReplied: number;
    isBest: boolean;
  }>;
  period: string;
  filters: { workspace: string | null };
}
```
