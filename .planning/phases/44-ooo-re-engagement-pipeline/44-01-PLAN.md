   ```
     workspaceSlug         String
     ebLeadId              Int?
     oooUntil              DateTime
     oooReason             String

     sentAt                DateTime?
     cancelledAt           DateTime?
     failureReason         String?
     createdAt             DateTime  @default(now())
     updatedAt             DateTime  @updatedAt

     @@unique([personEmail, workspaceSlug, status], name: "unique_pending_ooo")
     @@index([workspaceSlug, status])
     @@index([oooUntil])
     @@index([status])
   }
   ```
